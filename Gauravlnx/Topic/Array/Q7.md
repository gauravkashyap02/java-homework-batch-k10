package Array;

import java.util.Scanner;

public class Q7 {
public static void main(String args[]) {
Scanner sc = new Scanner(System.in);

        System.out.print("Enter size of array: ");
        int n = sc.nextInt();

        int arr[] = new int[n];

        System.out.print("Enter array elements: ");
        for(int i = 0; i < n; i++){
            arr[i] = sc.nextInt();
        }

        int countEven = 0, countOdd = 0;
        int sumEven = 0, sumOdd = 0;
        float avgEven , avgOdd;

        for(int i = 0; i < n; i++){
            if(i % 2 == 0){
                countEven++;
                sumEven += arr[i];
            }
        }
        avgEven = sumEven / countEven;
        System.out.println("Average of all even numbers is : "+avgEven);


        for(int i = 0; i < n; i++){
            if(i % 2 != 0){
                countOdd++;
                sumOdd += arr[i];
            }
        }
        avgOdd = sumOdd / countOdd;
        System.out.println("Average of all odd numbers is : "+avgOdd);





    }
}
